local function u(T, Z)
    local x = 1

    for k = 7, 0, -1.0 do
        local j = Z / _IMPORTED_P[k] % 2
        j = j - j % 1

        local Z_result = _IMPORTED_D(T, x + j * 256 + 256)

        x = x * 2 + Z_result

        if j ~= Z_result then
            while x < 0x100 do -- x held in R8.
                if x == 5 then
                    x = x * 2 + D(T, x)
                end
            end
            break
        end
    end

    return x % 256
end
